# Crawl_Facebook_Data_Toolbox
cd Crawl_Facebook_Data_Toolbox \
npm install webpack-dev-server \
npm audit fix --force \
npm run build-dev 
# Import toolbox
Go to your "Manage Extensions" of your Browser \
Load the folder "dist" in your project
